73:       def generate_upgraded_code   
74:         code = @gems.map do |name, version, data|
75:           version_string = (version ? "'#{version}'" : nil)
76:           source = data.delete(:source)
77:           
78:           data_string = nil
79:           unless data.empty?
80:             data_string = data.to_a.map {|k, v| ":#{k} => '#{v}'"}.join(", ")
81:           end
82:           
83:           # If we have a source, generate a call to +source+ then output the
84:           # gem call.  Otherwise, just generate the gem requirement.
85:           if source
86:             str = ["'#{name}'", version_string, data_string].compact.join(", ")
87:             "source '#{source}'\ngem #{str}"
88:           else
89:             str = ["'#{name}'", version_string, data_string].compact.join(", ")
90:             "gem #{str}"
91:           end
92:         end.join("\n")
93:       end
